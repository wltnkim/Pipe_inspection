CREATE TABLE pipe_normal (
 'seq' INT NOT NULL AUTO_INCREMENT,
 'img' LONGBLOB NOT NULL, 
 'timestamp' TIMESTAMP NOT NULL,
 PRIMARY KEY(seq)
)

CREATE TABLE pipe_crack (
 seq INT NOT NULL AUTO_INCREMENT,
 img LONGBLOB NOT NULL, 
 'timestamp' TIMESTAMP NOT NULL,
 PRIMARY KEY(seq)
)

CREATE TABLE pipe_hole (
 seq INT NOT NULL AUTO_INCREMENT,
 img LONGBLOB NOT NULL, 
 'timestamp' TIMESTAMP NOT NULL,
 PRIMARY KEY(seq)
)

CREATE TABLE pipe_data (
 seq INT NOT NULL AUTO_INCREMENT,
 img LONGBLOB NOT NULL, 
 timestamp TIMESTAMP NOT NULL DEFAULT now(3),
 class VARCHAR(5),
 PRIMARY KEY(seq)
)
